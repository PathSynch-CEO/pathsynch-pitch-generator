--- CREDIT CHECK PATTERN ---

// Atomic pattern (preferred — prevents double-spend):
const result = await checkAndDeductCredits(userId, CREDIT_COST, 'reason:context', { service: 'service_name' });
if (!result.allowed) {
  if (result.error === 'BILLING_TRANSACTION_FAILED') return res.status(503).json({ error: 'billing_error' });
  return res.status(402).json({ error: 'insufficient_credits', available: result.available });
}

const { sendProspectsToNemoClaw } = require('../services/prospectIntelService');

router.post('/prospect-intel/batch/:batchId/send-to-nemoclaw', requireAuth, async (req, res) => {
  const { batchId } = req.params;
  const userId = req.user.sub;
  const { prospectIds, options } = req.body;
  try {
    const result = await sendProspectsToNemoClaw(batchId, prospectIds, userId, options || {});
    return res.status(200).json({ success: true, ...result });
  } catch (err) {
    console.error('[ProspectIntel] sendToNemoClaw error:', err);
    return res.status(500).json({ error: err.message });
  }
});
